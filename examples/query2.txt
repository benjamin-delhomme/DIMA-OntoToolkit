#This query retrieves, for each article, the cognitive techniques it uses - grouped by DIMA phase (like Detect or Memorise) - along with how often each technique is applied, enabling structured comparison across narratives.

SELECT ?article (CONCAT(SUBSTR(?headline, 1, 30), "...") AS ?shortHeadline) ?techniqueType ?technique (COUNT(DISTINCT ?techniqueUsage) AS ?numTechniqueUsage)
WHERE {
  ?article scim:hasMotif ?motif .
  ?article scim:hasHeadline ?headline .
  ?motif scim:hasArgument ?argument .
  ?argument dima:usesTechnique ?techniqueUsage .
  ?techniqueUsage dima:instantiatesTechnique ?technique .
  ?technique rdfs:subClassOf ?techniqueType .
}
GROUP BY ?article ?shortHeadline ?techniqueType ?technique
ORDER BY ?article ?techniqueType ?technique'
